قرآن کے اس کہنے کا کیا مطلب ہے کہ ’اللہ سبحانہ و تعالیٰ نے شہد کی مکھی کو اپنے رب کے کشادہ راستے تلاش کرنا سکھایا ہے؟‘ فان فریش (von Frisch) ایک سائنسدان ہیں، انہیں 1973 میں نوبل انعام ملا جب انہوں نے جانوروں کے رویے خصوصاً شہد کی مکھی کے رابطے کے طریقے کو بیان کیا۔ اور آج ہمیں معلوم ہوا ہے کہ جب بھی کسی کارکن مکھی کو کوئی نیا پھول یا نیا باغ ملتا ہے جہاں سے وہ رس لے سکتی ہے، تو وہ جا کر اپنی ساتھی مکھیوں کو اس نئے پھول یا باغ کی صحیح سمت اور نقشہ بتاتی ہے... اسے ’بی ڈانس‘ (bee dance) کہا جاتا ہے۔ آج ہی ہمیں جدید فوٹو گرافی کے ذریعے معلوم ہوا ہے کہ یہ پیغام رسانی کیسے کی جاتی ہے۔

ان آیات میں جو صیغہ استعمال ہوا ہے وہ مادہ (female) کے لیے ہے، یعنی ’کُلی‘ یا ’فاسلُکی‘، جس کا مطلب ہے ’مادہ مکھی‘۔ پہلے ہم سمجھتے تھے کہ نر مکھی کارکن مکھی ہوتی ہے۔ کوئی تعجب نہیں کہ شیکسپیئر نے اپنے ڈرامے ’ہنری پنجم‘ میں لکھا کہ سپاہی مکھی (جو کہ نر ہے) جا کر بادشاہ کو رپورٹ کرتی ہے۔ آج ہمیں معلوم ہوا ہے کہ وہ بادشاہ کو نہیں بلکہ ملکہ کو رپورٹ کرتی ہیں اور کارکن مکھی نر نہیں بلکہ مادہ ہوتی ہے۔

لوگ سوچیں گے کہ قرآن کس طرح کی داستانوں کی کتاب ہے؟ قرآن چیونٹیوں کے بارے میں بات کر رہا ہے... چیونٹیاں آپس میں باتیں کر رہی ہیں، یہ تو کہانیوں کی کتاب لگتی ہے۔

آج ہمیں معلوم ہوا ہے کہ کوئی بھی جانور یا حشرہ جو انسان سے سب سے زیادہ مشابہت رکھتا ہے وہ چیونٹی ہے۔ چیونٹیاں اپنے مُردوں کو اسی طرح دفن کرتی ہیں جیسے انسان کرتے ہیں۔ ان کے پاس محنت کی تقسیم کا ایک پیچیدہ نظام ہے، جس میں منیجر، سپروائزر اور ورکرز وغیرہ ہوتے ہیں۔ وہ اکثر گپ شپ کے لیے ملتی ہیں۔ ان کے پاس رابطے کا ایک جدید طریقہ ہے۔ ان کے پاس منڈی (مارکیٹ) ہوتی ہے جہاں وہ سامان کا تبادلہ کرتی ہیں۔ اور جب وہ سردیوں کے لیے غلہ محفوظ کرتی ہیں، اور اگر بیج اگنے لگے، تو چیونٹیاں اس کے انکر (bud) کو کاٹ دیتی ہیں گویا وہ جانتی ہوں کہ اگنے سے غلہ سڑ جائے گا۔ اور مون سون میں اگر غلہ گیلا ہو جائے، تو وہ اسے خشک کرنے کے لیے دھوپ میں باہر لاتی ہیں۔ یقیناً آپ نے چیونٹیوں کو دھوپ میں غلہ لے جاتے دیکھا ہوگا گویا وہ جانتی ہوں کہ نمی جڑوں کے نظام کو فروغ دے گی جس سے غلہ سڑ جائے گا۔ چنانچہ قرآن 1400 سال پہلے چیونٹیوں کی اس ترقی یافتہ فطرت کے بارے میں بتاتا ہے۔
